# 2. 两数相加

Algorithm: 数学, 递归
Data Structure: Linked List
Difficulty: Medium
Last Review Time: September 5, 2022 1:41 PM
No: 2

给你两个 **非空** 的链表，表示两个非负的整数。它们每位数字都是按照 **逆序** 的方式存储的，并且每个节点只能存储 **一位** 数字。

请你将两个数相加，并以相同形式返回一个表示和的链表。

你可以假设除了数字 0 之外，这两个数都不会以 0 开头。

**示例 1：**

![https://cdn.nlark.com/yuque/0/2022/jpeg/22379150/1647169699129-836b92ef-d5df-4bb4-b12c-f6c0aef67c81.jpeg](https://cdn.nlark.com/yuque/0/2022/jpeg/22379150/1647169699129-836b92ef-d5df-4bb4-b12c-f6c0aef67c81.jpeg)

```
输入：l1 = [2,4,3], l2 = [5,6,4]
输出：[7,0,8]
解释：342 + 465 = 807.
```

**示例 2：**

```
输入：l1 = [0], l2 = [0]
输出：[0]
```

**示例 3：**

```
输入：l1 = [9,9,9,9,9,9,9], l2 = [9,9,9,9]
输出：[8,9,9,9,0,0,0,1]
```

**提示：**

- 每个链表中的节点数在范围 `[1, 100]` 内
- `0 <= Node.val <= 9`
- 题目数据保证列表表示的数字不含前导零

# 关键点

1. 设置进位inc
2. 两个链表的长度可能不相同
3. 另一个链表完全读完后，另一个链表依然需要与进位相加
4. 更新inc时，inc要在当前位数值计算完后改变

# 代码

### 方案1: 把l2合并到l1

```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */

// 核心理念 标记进位
// 两个链表的长度可能不同
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode* dummyhead = new ListNode(0, l1);
        ListNode* re = l1;
        int inc = 0;        // 进位

      // 在l1和l2都没有用完时
        while(l1 != nullptr && l2!= nullptr){
            if( (l1->val + l2->val + inc) >= 10){
                l1 -> val = (l1->val + l2->val + inc) % 10;
                inc = 1;    //关键点：inc要在当前位数值计算完后改变
            }
            else{
                l1 -> val = l1->val + l2->val + inc;
                inc = 0;
            }
            l1 = l1 -> next;
            l2 = l2 -> next;
        }

      // 进位为0
      // 如果l2为加完，l1直接指向l2即可 但是因为l1走完上述循环为nullptr，因此需要设置一个新的re指针等于l1，走到最后一个位置，连接l2
        if(inc == 0){
            if(l1 == nullptr && l2 != nullptr){
                while(re->next){
                    re = re -> next;
                }
                re -> next = l2;
            }
            else{
                return dummyhead -> next;
            }
        }

      // 进位为1
        else{
          // l1 l2 都为空需要添加一个新节点存放inc
            if(l1 == nullptr && l2 == nullptr){
                ListNode* node = new ListNode(inc);
                while(re->next){
                    re = re -> next;
                }
                re -> next = node;
            }
          // l2 + inc 添加到l1链表的后面
          // l1，l2 都为空可能还存在inc == 1的情况
            else if (l1 == nullptr && l2 != nullptr){
                while(re->next){
                    re = re -> next;
                }
                re -> next = l2;
                while(l2){
                    if((l2->val + inc) >= 10){
                        l2 -> val = (l2->val + inc) % 10;
                        inc = 1;
                    }
                    else{
                        l2 -> val = (l2->val + inc) % 10;
                        inc = 0;
                    }
                    l2 = l2 -> next;
                }
                if(l1 == nullptr && l2 == nullptr && inc == 1){
                    ListNode* node = new ListNode(inc);
                    while(re->next){
                        re = re -> next;
                    }
                    re -> next = node;
                }
            }

          // l1链表的剩余部分与inc相加
          // l1，l2 都为空可能还存在inc == 1的情况
            else if (l1 != nullptr && l2 == nullptr){
                while(l1){
                    if((l1->val + inc) >= 10){
                        l1 -> val = (l1->val + inc) % 10;
                        inc = 1;
                    }
                    else{
                        l1 -> val = (l1->val + inc) % 10;
                        inc = 0;
                    }
                    l1 = l1 -> next;
                }
                if(l1 == nullptr && l2 == nullptr && inc == 1){
                    ListNode* node = new ListNode(inc);
                    while(re->next){
                        re = re -> next;
                    }
                    re -> next = node;
                }
            }
        }

        return dummyhead -> next;
    }
};
```

### 方案2: 开辟新的节点存放相加结果

```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode* head = new ListNode(0);
        ListNode* dummyhead = head;
        int inc = 0;
        while( l1!=nullptr && l2 != nullptr )
        {
            int temp = l1->val + l2->val + inc;
            if(temp >= 10)
            {
                inc = 1;
                temp = temp%10;

            }
            else{
                inc = 0;
            }
            ListNode* node = new ListNode(temp);
            head->next = node;
            head = head->next;
            l1 = l1->next;
            l2 = l2->next;
        }

        if(l1!= nullptr){
            while(l1 != nullptr){
                ListNode* node = new ListNode((l1->val + inc)%10);
                if (l1->val + inc >= 10)
                {
                    inc = 1;
                }
                else{
                    inc = 0;
                }
                head->next = node;
                head = head->next;
                l1 = l1->next;
            }
        }
        else if(l2 != nullptr){
            while(l2 != nullptr){
                ListNode* node = new ListNode((l2->val + inc)%10);
                if (l2->val + inc >= 10)
                {
                    inc = 1;
                }
                else{
                    inc = 0;
                }

                head->next = node;
                head = head->next;
                l2 = l2->next;
            }
        }

        if(inc == 1)
        {
            ListNode* node = new ListNode(1);
            head->next = node;
        }

        return dummyhead->next;
    }
};
```
