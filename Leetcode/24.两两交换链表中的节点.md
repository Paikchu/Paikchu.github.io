ListNode* swapPairs(ListNode* head) {
      if(head == nullptr)
          return head;
      ListNode* dummyhead = new ListNode(0);
      dummyhead -> next = head;
      int count = 0;
      // {head:1, fast:2, 3, 4}

       ListNode* temp = dummyhead;
